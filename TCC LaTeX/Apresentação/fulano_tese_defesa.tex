\section{Introdução}

\frame{
    \frametitle{Estabilidade de Sistemas Dinâmicos}

    Considere o sistema linear
    \begin{equation}
        \dot{x}=Ax,~x\in \reais^n
        \label{eq:00}
    \end{equation}


    \begin{teorema}[Lyapunov]
    Estabilidade (Lyapunov): O sistema linear dado por \bref{eq:00} é estável (trajetórias iniciando em
    qualquer ponto convergem para $x=0$) se e somente se
    existir $P=P'\in \reais^{n \times n}$ tal que
    \[
    A'P + PA < 0;~P>0
    \]
    \end{teorema}

    \begin{itemize}
    \item Definindo-se um critério linear para os elementos da matriz $P$, as desigualdades $A'P + PA < 0;~P>0$
    podem ser colocadas na forma do seguinte problema
          otimização.
    \end{itemize}

}

\frame{
    \frametitle{Programação Semidefinida}

    \begin{gather}
    \nonumber \min_{x \in \reais^{r}} c'x \\
    \label{C1:eq:AP}\mbox{tal que} \quad F(x) \triangleq F_0 + x_1 F_1 + x_2 F_2 + \cdots + x_r F_r > 0
    \end{gather}
    sendo que $c \in \reais^{r}$, $F_i=F_i' \in \reais^{q \times q},~i=1,\ldots,r$ e $x \in \reais^r$ são
    as variáveis de decisão do problema.

   \begin{itemize}
    \item Desigualdade matricial linear - LMI

    \item Existem programas computacionais especializados (algoritmos
    convergem em tempo polinomial) na resolução de LMIs: LMI Control Toolbox
    (MatLab), SeDuMi, etc.

    \item Inúmeros problemas de controle podem ser formulados como LMIs.
    %\item Formular um problema em termos de LMIs equivale a resolver     o problema!
    \end{itemize}

}

\frame
{
    \frametitle{Sistemas Lineares Incertos}

    Sistema linear incerto com representação politópica:
    \begin{equation}
    \label{eq:01}
        \dot{x}=\Aal x,\qquad \Aal \in \Afr
    \end{equation}

    \[
    \Afr \triangleq \left\{\Aal : \Aal= \sum_{i=1}^{N}\alpha_i A_i,~~\alpha \in \simplex\right\}
    \]
    \[
    \simplex = \left\{\alpha \in \reais^N: \sum_{i=1}^{N}\alpha_i=1,~\alpha_i \geq 0,~i=1,\ldots,N \right\}
    \]

    $\simplex$ - {\em simplex unitário}.

}

\frame
{
    \frametitle{Estabilidade Robusta}

    \begin{alertblock}{Questão}
    Como decidir sobre a estabilidade robusta de $\Aal ~\forall \alpha \in \simplex$ por meio de um número
    finito de testes (LMIs) ?
    \end{alertblock}

    \begin{teorema}[Lyapunov]
    O sistema linear dado por \bref{eq:01} é estável  se e somente se
    existir $\Pal=\Pal'\in \reais^{n \times n}$ tal que
    \be
    \Aal'\Pal + \Pal\Aal < 0;~\Pal>0,~\forall \alpha \in \simplex
    \label{eq:02}
    \ee
    \end{teorema}

    %\begin{itemize}
    %\item As desigualdades em \bref{eq:02} são LMIs dependentes de parâmetros.
    %\end{itemize}

}

\frame
{
    \frametitle{LMIs dependentes de parâmetros}


    %\begin{itemize}
    %\item As desigualdades em \bref{eq:02} são LMIs dependentes de parâmetros.
    %\end{itemize}
    \begin{block}{Comentários}

    \begin{itemize}
    \item $\Pal$ não possui uma estrutura particular.

    \item Infinitos números de testes.

    \item LMIs dependentes de parâmetros.

    \item Saída: Arbitrar uma estrutura para $\Pal$, ao preço de um possível conservadorismo
    na solução.

    %\item Estabilidade quadrática, funções de Lyapunov com dependência afim nos parâmetros

    \end{itemize}
    \end{block}

}

\frame
{
    \frametitle{Resultados da Literatura}

    \begin{block}{Resultados Anteriores}

    \begin{itemize}
    \item Estabilidade quadrática ($\Pal=P$) $\rightarrow$ simples mas conservadora.

    \item Funções de Lyapunov com dependência afim nos parâmetros ($\Pal = \sum \alpha_i P_i$)
          $\rightarrow$ [FAG96], [GdOH98], [RP01] (ainda) conservadoras.

    \item Funções de Lyapunov com dependência polinomial nos parâmetros $\rightarrow$ resultados não
    conservadores.

    \end{itemize}
    \end{block}

    \begin{alertblock}{Objetivo}
    O objetivo desta tese é apresentar uma metodologia sistemática para encontrar
           soluções de LMIs dependentes de parâmetros que pertencem ao simplex unitário.
    \end{alertblock}


}

\section{Resultados}

\frame
{
    \frametitle{Funções de Lyapunov lineares}

    Considere a a função de Lyapunov $v(x)=x'\Pal x$ com
    \[
    \Pal=\sum_{i=1}^N \alpha_i P_i,\qquad \alpha \in \simplex
    \]
    Aplicando as definições de $\Aal$  e $\Pal$, tem-se
    \begin{multline}
    \Aal'\Pal + \Pal\Aal= \sum_{i=1}^N \alpha_i^2(\underbrace{A_i'P_i + P_iA_i}_{T_i}) + \\
    \sum_{i=1}^{N-1} \sum_{j=i+1}^N \alpha_i\alpha_j(\underbrace{A_i'P_j + P_jA_i + A_j'P_i + P_iA_j}_{T_{ij}})
    \label{eq:03}
    \end{multline}
    Condição suficiente: $P_i>0$, $T_i<0$ e $T_{ij}<0$.
}

\frame
{
    \frametitle{Suficiência $\rightarrow$ Necessidade ?}

    \begin{alertblock}{Questão}
    A partir de \bref{eq:03}, é possível obter condições que também sejam necessárias?
    \end{alertblock}

    \begin{teorema}[Pólya]
    Seja $F(\alpha) \triangleq F(\alpha_1,\alpha_2,\ldots,\alpha_N)$ um polinômio homogêneo real que é positivo
    $\forall \alpha \in \simplex$. Então para um $d \in \inteiros$ suficientemente grande, o produto
    \[
    (\alpha_1+\alpha_2+\ldots+\alpha_N)^d F(\alpha)
    \]
    tem todos os seus coeficientes estritamente positivos.
    \end{teorema}


}

\frame
{
    \frametitle{Resultado}


    \begin{teorema}[Oliveira \& Peres, LAA (2005)]
    \label{OP:05c}
    Uma matriz de Lyapunov com dependência afim nos parâmetros garante a estabilidade robusta de
    $\Afr$ se e somente se existirem matrizes simétricas definidas positivas $P_i,~i=1,\ldots,N$ e um $d$ suficientemente
    grande tais que os coeficientes do polinômio homogêneo
    \[
    (\alpha_1+\ldots+\alpha_N)^d \bigl(\Aal'\Pal + \Pal \Aal \bigr)
    \]
    sejam todos definidos negativos.
    \end{teorema}


}

\frame
{
  \frametitle{Estudo de caso $N=2$}

    Considere as condições do Teorema~\ref{OP:05c} aplicadas
    a um politopo $\mathcal{A}$ de $N=2$ vértices:
    \[
    T_{H_1} = A_1'P_1 + P_1 A_1 ; \qquad
    T_{H_2}  = A_2'P_2 + P_2 A_2
    \]
    \[
    T_{H_{12}} =
    A_1'P_2 + P_2 A_1+ A_2'P_1 + P_1 A_2
    \]

    \implica Para $d=0$, as LMIs
    \be
    T_{H_1} < 0 ; \qquad T_{H_2} < 0
    \label{th1th2}
    \ee
    que são condições necessárias e
    \be
    T_{H_{12}} < 0
    \label{th12}
    \ee

    \implica Para $d=1$, as LMIs são \bref{th1th2} e
    \be
    T_{H_1} +T_{H_{12}} < 0 ; \qquad
    T_{H_2} + T_{H_{12}} < 0
    \label{td1}
    \ee
}

\frame
{
    \frametitle{Funções de Lyapunov Polinomiais}


    De acordo com [Bli04], LMIs dependentes de parâmetros escalares podem necessitar
    de soluções polinomiais nos parâmetros.

    \begin{block}{Proposta}
    Usar funções de Lyapunov polinomiais homogêneas nos parâmetros
    \be
    P_g(\alpha) = \sum_{j=1}^{J(g)}
    \alpha_1^{k_1}\alpha_2^{k_2}\cdots
    \alpha_N^{k_N}
    P_{\Kfr_j (g)} ~ ; \quad
    k_1k_2\cdots k_N = \Kfr_{j} (g)
    \label{eq:12}
    \ee
    \end{block}

}

\frame
{
    \frametitle{Resultado}

    \begin{teorema}[Oliveira \& Peres, SCL (2006)]
    \label{OP:06a}
    Se existir uma matriz de Lyapunov polinomial homogênea de grau arbitrário nos parâmetros
    $P_g(\alpha)>0$ tal que
    \be
    \Aal'P_g(\alpha) + P_g(\alpha) \Aal<0
    \label{eq:scl}
    \ee
    então o politopo $\Afr$ é estável.
    \end{teorema}
    \begin{itemize}
    \item A parametrização para \bref{eq:scl} pode ser encontrada na tese.
    \item Os resultados mostram que o conservadorismo das condições diminui
    à medida que o grau da função de Lyapunov aumenta.
    \end{itemize}


}

\frame
{
    \frametitle{Funções Polinomiais \& Pólya}

    \begin{teorema}
    \label{OP:05f}
    Uma função de Lyapunov polinomial homogênea de grau arbitrário nos parâmetros
    $P_g(\alpha)$ garante a estabilidade de $\Afr$ se e somente se existir um $d$ suficientemente
    grande tal que
    \begin{gather*}
    (\alpha_1+\ldots+\alpha_N)^d P_g(\alpha)>0;\\
    (\alpha_1+\ldots+\alpha_N)^d(\Aal'P_g(\alpha) + P_g(\alpha) \Aal)<0;
    %\label{eq:ieee}
    \end{gather*}
    \end{teorema}
    \begin{itemize}
    \item %A condição fornece dois graus de liberdade para controlar o compromisso
    %entre resultados precisos e o esforço computacional.
    $g$ e $d$.
    \item Não há perda de generalidade quando se consideram funções de Lyapunov polinomiais homogêneas ao invés
    de funções polinomiais gerais.
    \item Fato interessante: Com o resultado do Teorema~\ref{OP:05f} é possível provar a convergência assintótica
    do Teorema~\ref{OP:06a}.
    \end{itemize}


}

\frame
{
    \frametitle{Aplicações Imediatas}

    A flexibilidade da metodologia permite tratar outras LMIs dependentes de parâmetros.

    \begin{teorema}[Condição de estabilidade estendida]
    \label{OP:05f_fins}
    Uma função de Lyapunov polinomial homogênea de grau arbitrário nos parâmetros
    $P_g(\alpha)$ garante a estabilidade de $\Afr$ se e somente se existir um $d$ suficientemente
    grande e uma matriz polinomial homogênea $X_g(\alpha)$ tais que
    \begin{gather*}
    (\alpha_1+\ldots+\alpha_N)^d P_g(\alpha)>0;\\
    (\alpha_1+\ldots+\alpha_N)^d \bigl(Q_g(\alpha) + X_g(\alpha)\mathcal{B}(\alpha) + \mathcal{B}(\alpha)'X_g(\alpha)'\bigr)<0
    \end{gather*}
    com
    \[
    \left[\begin{array}{cc}0 & P_g(\alpha) \\ P_g(\alpha) & 0 \end{array}\right],\quad
    \mathcal{B}(\alpha) = [\Aal~~-I]
    \]
    \end{teorema}
}

\section{Exemplos Numéricos}

\frame
{
    \frametitle{Métodos Polinomiais}

    Para o caso de estabilidade robusta, as condições propostas na tese são comparadas
    com outros métodos da literatura que também são suficientes e assintoticamente necessários.

    \begin{block}{Chesi et al. (CGTV05)}
    Garante a estabilidade robusta por meio da existência de uma função
        de Lyapunov polinomial homogênea. Transforma o problema na analise da positividade de uma
        matriz de Gram (representação completa de matriz quadrada), obtendo relaxações LMIs convergentes.
    \end{block}
}

\frame {
    \frametitle{Métodos Polinomiais}

    \begin{block}{P.-A. Bliman (Bli04)}
    Garante a estabilidade robusta de um sistema linear com incertezas na forma afim por
    meio da existência de uma matriz de Lyapunov polinomial geral. Usa o Lema de KYP para
    obter condições LMIs assintoticamente necessárias. Pode ser adaptado para o caso politópico,
    resultando na análise de $N$ sistemas afins com $N-1$ parâmetros incertos cada.
    \end{block}
}

\frame {
    \frametitle{Métodos Polinomiais}

    \begin{block}{Henrion et al. (HAPL04)}
    Por meio do critério de estabilidade de Hermite, o problema de estabilidade robusta pode
    ser escrito na forma de um problema de otimização envolvendo função objetivo
    e restrições polinomiais. O software de otimização global de polinômios
    ``Gloptipoly'' é usado para determinar se o polinômio é positivo, e conseqüentemente, garantir
    a estabilidade robusta. O problema de otimização é resolvido por meio de uma seqüencia
    de condições LMIs assintoticamente convergentes (teoria de momentos, Lasserre 2001).
    \end{block}
}

\section{Conclusões}

\frame
{

    \begin{exampleblock}{Concluindo}
        \begin{enumerate}
            \item<1-| alert@1>
                Relaxações LMI assintoticamente exatas foram apresentadas para encontrar soluções de
                LMIs dependentes de parâmetros ligadas à análise robusta de sistemas lineares com incertezas politópicas.
            \item<2-| alert@2>
                As relaxações exploram funções de Lyapunov com dependência polinomial homogênea nos parâmetros.
            \item<3-| alert@3>
                O Teorema de Pólya é usado para garantir a convergência das condições propostas.
            \item<4-| alert@4>
                Os métodos propostos apresentam bons resultados quando comparados com os outros métodos da literatura.
            \item<5-| alert@5>
                Outros problemas de análise e controle escalonado podem ser abordados de maneira similar.
    \end{enumerate}
  \end{exampleblock}

}
